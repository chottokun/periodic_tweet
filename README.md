# periodic_tweet
ツイートする文章を書いたファイルからランダムに抽出し定期的にツイートする。

# 設定
configure.ini に twitter APIとツイートの最後に追加する文章を記載する。

        [TWITTER]
        # API
        CONSUMER_KEY =
        CONSUMER_SECRET =
        ACCESS_TOKEN =
        ACCESS_TOKEN_SECRET =
        # tweet words to add
        WORDS =
